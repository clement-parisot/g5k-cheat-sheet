\begin{lstlisting}
oarsub -t deploy -l {"type='kavlan'"}/vlan=1+nodes=2\
walltime=2 -I
\end{lstlisting}

\textbf{Deploy}
\begin{lstlisting}
kadeploy3 -f $OAR_NODEFILE -e env -k --vlan `kavlan -V`
\end{lstlisting}

\textbf{Find out in which vlan is a node}
\begin{lstlisting}
kavlan -g -m node.fqdn.fr
\end{lstlisting}

\textbf{List nodes (kavlan fqdn of a reservation)}
\begin{lstlisting}
kavlan -l -j jobid
\end{lstlisting}
